Username Generator [TEAM WORK]


//Username Generator
      
Console.Write("What is your name: ");
String name = Console.ReadLine();
      
Console.Write("How old are you: ");
int age = Convert.ToInt32(Console.ReadLine());
      
int maxPrefixes = 5;
      
Random random = new Random();
int randomNumberPrefix = random.Next(1,maxPrefixes+1);
      
//String Declaration
String namePrefixOne = "Mr.";
String namePrefixTwo = "Light";
String namePrefixThree = "Shadow";
String namePrefixFour = "Dark";
String namePrefixFive = "Gotiniq";
      
Console.Write("What is your username symbol: ");
char prefixSymbol = Convert.ToChar(Console.ReadLine());
      
switch (randomNumberPrefix)
{
       case 1:
           Console.WriteLine("Your username is " + prefixSymbol + namePrefixOne + name + age);
           break;
       case 2:
           Console.WriteLine("Your username is " + prefixSymbol + namePrefixTwo + name + age);
           break;
       case 3:
           Console.WriteLine("Your username is " + prefixSymbol + namePrefixThree + name + age);
           break;
       case 4:
           Console.WriteLine("Your username is " + prefixSymbol + namePrefixFour + name + age);
           break;
       case 5:
           Console.WriteLine("Your username is " + prefixSymbol + namePrefixFive + name + age);
           break;
}
      
Console.ReadKey();

Basic Calcualator


Console.Write("Enter first number: ");
double num1 = Convert.ToDouble(Console.ReadLine());
      
Console.Write("Enter second number: ");
double num2 = Convert.ToDouble(Console.ReadLine());
      
Console.Write("Enter a operation: ");
char operation = Convert.ToChar(Console.ReadLine());
      
switch (operation)
{
       case '+':
           Console.WriteLine(num1 + num2);
           break;
       case '-':
           Console.WriteLine(num1 - num2);
           break;
       case '':
           Console.WriteLine(num1 num2);
           break;
       case '/':
           Console.WriteLine(num1 / num2);
           break;
       default:
          Console.WriteLine("Invalid operation");
          return;
}

Console.ReadKey();
